CLASS XSQLTable INHERIT SQLTable
    PROTECT cUpdate AS STRING
    PROTECT cUnique AS STRING 
    PROTECT cFileNameAbbrv AS STRING

// --- Métodos de Leitura/Gravação Automática ---

METHOD FieldPut(nField, uValue) CLASS XSQLTable
    // Atualiza na memória do servidor
    SUPER:FieldPut(nField, uValue)
    // Se quiser que a gravação seja automática (estilo DBF), 
    // descomente a linha abaixo (pode impactar performance em grids):
    // SELF:UpdateRow( { {SELF:FieldName(nField), uValue} }, "FQ", 0, SELF:Unique_ID )
RETURN uValue

METHOD UpdateRow(aSave, cField, nClient, cQ) CLASS XSQLTable
    LOCAL cStatement, cWhere AS STRING
    LOCAL i AS DWORD
    
    Default(@cField, "FQ")
    Default(@nClient, 0)
    Default(@cQ, SELF:Unique_ID)
    
    // Auto-insere TimeStamp se o campo existir
    IF SELF:FieldPos("FDT") > 0 .AND. AScan(aSave, {|x| x[1] == "FDT"}) == 0
        AAdd(aSave, {"FDT", SELF:XReplDT()})
    ENDIF

    cWhere := IIF(nClient > 0, cField + "=" + LTrim(Str(nClient)), "FQ='" + cQ + "'")
    
    cStatement := "UPDATE " + SELF:Name + " SET "
    FOR i := 1 TO ALen(aSave)
        cStatement += aSave[i, 1] + "=" + SELF:FormatValue(aSave[i, 2]) + IIF(i < ALen(aSave), ", ", "")
    NEXT
    cStatement += " WHERE " + cWhere
    
RETURN SELF:Execute(cStatement)

// --- Utilitários de Conversão (Antigas Funções Globais) ---

METHOD FormatValue(uValue AS USUAL) CLASS XSQLTable
    LOCAL cType := ValType(uValue) AS STRING
    DO CASE
        CASE cType == "C" ; RETURN "'" + RTrim(uValue) + "'"
        CASE cType == "L" ; RETURN IIF(uValue, "1", "0")
        CASE cType == "D" ; RETURN "'" + DToS(uValue) + "'"
        CASE cType == "N" ; RETURN LTrim(Str(uValue, 20, 5))
        OTHERWISE         ; RETURN "' '"
    ENDCASE

METHOD Execute(cSQL AS STRING) CLASS XSQLTable
    LOCAL oStmt AS SQLStatement
    oStmt := SQLStatement{cSQL, GLOoCONNECTION}
RETURN oStmt:Execute()

// --- Métodos de Apoio ao Fluxo (TimeStamp/ID) ---

METHOD XReplDT() CLASS XSQLTable
    SELF:cUpdate := DToS(Today()) + SubStr(Time(), 1, 5)
RETURN SELF:cUpdate

ACCESS Unique_ID CLASS XSQLTable
    IF Empty(SELF:cUnique) .AND. SELF:FieldPos("FQ") > 0
        SELF:cUnique := SELF:FieldGet(#FQ)
        IF Empty(SELF:cUnique)
            SELF:cUnique := SELF:XReplQ(SELF:Name)
            SELF:FieldPut(#FQ, SELF:cUnique)
        ENDIF
    ENDIF
RETURN SELF:cUnique

METHOD XReplQ(cFile AS STRING) CLASS XSQLTable
    STATIC nRand := 0 AS INT
    nRand++
RETURN PadR(cFile, 3, "_") + CHR(Val(SubStr(Time(), 7, 2)) + 63) + Right(Str(Rand(nRand), 12, 9), 7)
