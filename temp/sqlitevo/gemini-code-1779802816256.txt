METHOD Append() CLASS XSQLTable
    // O SQLTable padrão do VO não tem um Append automático eficiente.
    // Este método prepara um INSERT vazio para garantir que o registro exista.
    LOCAL cStatement AS STRING
    cStatement := "INSERT INTO " + SELF:Name + " (FQ) VALUES ('" + SELF:XReplQ(SELF:Name) + "')"
    RETURN SELF:Execute(cStatement)

METHOD AppendNew() CLASS XSQLTable
    // Garante que o novo registro tenha os valores padrão (NOT NULL)
    IF SELF:Append()
        SELF:GoBottom() // Posiciona no novo registro
        // Aqui você pode adicionar lógica para preencher campos default
        RETURN TRUE
    ENDIF
RETURN FALSE

METHOD Delete() CLASS XSQLTable
    // Substitui o comando DELETE do DBF pelo DELETE SQL
    // Usa o Unique_ID (FQ) para garantir que apenas o registro atual seja deletado
    RETURN SELF:DeleteRow("FQ", 0, SELF:Unique_ID)

METHOD Count() CLASS XSQLTable
    // Substitui o RecCount()
    LOCAL oSelect AS XSQLSelect
    LOCAL nRet AS INT
    oSelect := XSQLSelect{"SELECT count(*) FROM " + SELF:Name, SELF:oConn}
    nRet := oSelect:FieldGet(1)
    oSelect:Close()
RETURN nRet

METHOD Seek(sField, uValue, lExact) CLASS XSQLTable
    // O Seek nativo de DBF não existe no SQL.
    // Este método faz um GoTop e navega até encontrar o valor.
    LOCAL lFound := FALSE AS LOGIC
    Default(@lExact, TRUE)
    SELF:GoTop()
    WHILE !SELF:EoF
        IF SELF:FieldGet(sField) == uValue
            lFound := TRUE
            EXIT
        ENDIF
        SELF:Skip(1)
    ENDDO
RETURN lFound
